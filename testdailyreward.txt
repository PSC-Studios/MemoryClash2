  <script>
    const wheel = document.getElementById("wheel");
    const result = document.getElementById("result");
    const spinBtn = document.getElementById("spinBtn");

    const rewards = [5, 10, 20, 25, 50, 75, 100, 500]; // Jackpot = 500
    const colors = ["#f8b195", "#f67280", "#c06c84", "#6c5b7b", "#355c7d", "#99b898", "#feceab", "#ff847c"];

    const segmentAngle = 360 / rewards.length;

    // Erstelle Segmente
    rewards.forEach((value, i) => {
      const segment = document.createElement("div");
      segment.className = "segment";
      segment.style.transform = `rotate(${i * segmentAngle}deg) skewY(${90 - segmentAngle}deg)`;
      segment.style.background = colors[i % colors.length];
      segment.innerText = value + "üí∞";
      wheel.appendChild(segment);
    });

    let spinning = false;

    spinBtn.addEventListener("click", () => {
      if (spinning) return;

      if (localStorage.getItem("lastSpin")) {
        const lastSpin = new Date(localStorage.getItem("lastSpin"));
        const now = new Date();
        const diff = now - lastSpin;
        if (diff < 86400000) {
          result.innerText = "Komm morgen wieder!";
          return;
        }
      }

      spinning = true;
      const spins = 5 + Math.floor(Math.random() * 5); // 5-9 Umdrehungen
      const index = Math.floor(Math.random() * rewards.length);
      const angle = 360 * spins + (360 - index * segmentAngle - segmentAngle / 2);

      wheel.style.transition = "transform 4s ease-out";
      wheel.style.transform = `rotate(${angle}deg)`;

      setTimeout(() => {
        const won = rewards[index];
        result.innerText = `Du hast ${won} Coins gewonnen! ü•≥`;
        const coins = parseInt(localStorage.getItem("coins") || "0");
        localStorage.setItem("coins", coins + won);
        localStorage.setItem("lastSpin", new Date());
        spinning = false;
      }, 4000);
    });
  </script>
